-	script	LoginEvent	-1,{
OnPCLoginEvent:
	// Initialize
	if(!isInitialize){
		isInitialize = 1;
	}
	
	// Newbie characters will wear summer costume
	if(Class == 0)
	sc_start SC_SUMMER,-1,0;
	// Always return to lobby
	else
	addtimer 100,strnpcinfo(0) + "::OnReturnToLobby";

	end;
	
OnReturnToLobby:
	warp "nif_dun01",145,140;
	
	CurrentPoints();
	
	SkillPoint = 0;
	
	Zeny = 1000000000;
	
	callfunc("AutoSellLockSkill");
	
	setpcblock PCBLOCK_SKILL,true;
	
	end;
}
